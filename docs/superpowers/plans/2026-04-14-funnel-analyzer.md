# Funnel Analyzer API — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a local REST API (`POST /analyze`) that receives an `account_id` and an AI prompt, fetches all conversations from Supabase, classifies them into funnel stages deterministically, computes funnel metrics, and returns JSON + a markdown report generated by `gpt-4o-mini`.

**Architecture:** Express server → parallel fetch (Supabase data + LLM prompt parse with cache) → filter professional/personal → stage detection via regex → metrics computation → LLM report generation → JSON response.

**Tech Stack:** Node.js 18+, Express 4, OpenAI SDK 4, Jest 29, native `fetch` (Node 18 built-in), native `crypto` (MD5), native `fs` (cache).

```js
const crypto = require('crypto');
const fs = require('fs');
const path = require('path');

const CACHE_DIR = process.env.CACHE_DIR || path.join(__dirname, '..', 'cache');

function ensureDir() {
  if (!fs.existsSync(CACHE_DIR)) fs.mkdirSync(CACHE_DIR, { recursive: true });
}

function hashPrompt(text) {
  return crypto.createHash('md5').update(text).digest('hex');
}

function get(hash) {
  ensureDir();
  const file = path.join(CACHE_DIR, `${hash}.json`);
  if (!fs.existsSync(file)) return null;
  return JSON.parse(fs.readFileSync(file, 'utf8'));
}

function set(hash, data) {
  ensureDir();
  fs.writeFileSync(path.join(CACHE_DIR, `${hash}.json`), JSON.stringify(data, null, 2));
}

function list() {
  ensureDir();
  return fs.readdirSync(CACHE_DIR)
    .filter(f => f.endsWith('.json'))
    .map(f => {
      const stat = fs.statSync(path.join(CACHE_DIR, f));
      return { hash: f.replace('.json', ''), created_at: stat.birthtime.toISOString(), size_bytes: stat.size };
    });
}

function del(hash) {
  const file = path.join(CACHE_DIR, `${hash}.json`);
  if (fs.existsSync(file)) fs.unlinkSync(file);
  return true;
}

module.exports = { hashPrompt, get, set, list, del };
```

```js
const DEFAULT_PROCEDURE = /fastlifting|full[\s.-]?face|deep[\s.-]?plane|lifting|minilipo|abdominoplastia|mastopexia|ginecomastia|lipoescultura/i;
const DEFAULT_CLINICAL = /papada|flacid|bigode|mandíbula|mandibula|maçã|olhos?|pálpebra|palpebra|sobrancelh|pescoço|pescoco|rosto|face|facial|rugas?|rejuvenescimento|estétic|estetic/i;
const COMMERCIAL = /consulta|investimento|agendamento|se interessou|procedimento|cirurgia|recuperação|anestesia/i;
const PRICE = /R\$\s*[\d.,]+\s*(mil|k)?/i;

function buildRegex(terms) {
  if (!terms || terms.length === 0) return null;
  const escaped = terms.map(t => t.replace(/[.*+?^${}()|[\]\\]/g, '\\$&'));
  return new RegExp(escaped.join('|'), 'i');
}

function isProfessional(messages, stageConfig = {}) {
  const clinicalRx = buildRegex(stageConfig.clinical_terms) || DEFAULT_CLINICAL;
  const procedureRx = buildRegex(stageConfig.procedure_terms) || DEFAULT_PROCEDURE;

  const outbound = messages.filter(m => m.direction === 'outbound');
  const inbound = messages.filter(m => m.direction === 'inbound');

  // Always PERSONAL overrides
  if (outbound.length === 0) return false;
  if (messages.length <= 1 && (messages[0]?.content_text || '').length < 5) return false;
  const senderKeys = new Set(messages.map(m => `${m.direction}_${m.sent_by}`));
  if (senderKeys.size <= 1) return false;

  // Tier 1: IA message present
  if (messages.some(m => m.sent_by === 'IA')) return true;

  // Tier 2: human outbound has professional language
  if (outbound.some(m => {
    const t = m.content_text || '';
    return procedureRx.test(t) || clinicalRx.test(t) || COMMERCIAL.test(t) || PRICE.test(t);
  })) return true;

  // Tier 3: patient mentioned clinical topic + substantive human response
  const patientClinical = inbound.some(m => {
    const t = m.content_text || '';
    return (clinicalRx.test(t) || procedureRx.test(t)) && t.length > 10;
  });
  const substantiveResponse = outbound.some(m => (m.content_text || '').length > 30);
  return patientClinical && substantiveResponse;
}

module.exports = { isProfessional };
```

```js
const STAGE_ORDER = [
  'CONTATO_INICIAL',
  'SAUDACAO',
  'APRESENTACAO_TECNICA',
  'QUEIXA_DECLARADA',
  'TRIAGEM_CLINICA',
  'INVESTIMENTO',
  'OBJECAO',
  'CAPTURA_CONTATO',
  'HANDOFF_HUMANO'
];

function buildRx(terms) {
  if (!terms || terms.length === 0) return null;
  return new RegExp(terms.map(t => t.replace(/[.*+?^${}()|[\]\\]/g, '\\$&')).join('|'), 'i');
}

function classifyTrack(outboundIA, outboundHuman) {
  if (outboundIA > 0 && outboundHuman > 0) return 'hybrid';
  if (outboundIA > 0) return 'pure_ia';
  if (outboundHuman > 0) return 'pure_human';
  return 'no_outbound';
}

function detectStages(messages, stageConfig) {
  const cfg = stageConfig || {};
  const greetingRx = buildRx(cfg.greeting_patterns) || /^(oi\b|olá|bom dia|boa tarde|boa noite)/i;
  const procedureRx = buildRx(cfg.procedure_terms) || /fastlifting|full[\s.-]?face|lifting/i;
  const clinicalRx = buildRx(cfg.clinical_terms) || /papada|flacid|bigode|olhos?|sobrancelh|pescoço|mandíbula/i;
  const prevTreatRx = buildRx(cfg.previous_treatment_terms) || /já (realizou|fez|fiz)|botox|preenchimento|tratamento anterior/i;
  const priceRx = new RegExp(cfg.price_pattern || 'R\\$\\s*[\\d.,]+\\s*(mil|k)?', 'i');
  const objectionRx = buildRx(cfg.objection_terms) || /longe|medo|caro|distância|não tenho dinheiro/i;
  const contactRx = buildRx(cfg.contact_capture_terms) || /whatsapp|minha equipe|equipe vai entrar/i;

  const events = [];
  const stagesHit = new Set();
  let hasIA = false;

  const push = (stage, sender, msg) => {
    if (stagesHit.has(stage) && stage !== 'OBJECAO') return;
    stagesHit.add(stage);
    events.push({
      stage,
      sender,
      timestamp: (msg.created_at || '').substring(0, 16),
      preview: (msg.content_text || '').substring(0, 80)
    });
  };

  for (const m of messages) {
    const txt = (m.content_text || '').trim();
    const isIA = m.sent_by === 'IA';
    const isHumanOut = m.direction === 'outbound' && !isIA;
    const isIn = m.direction === 'inbound';
    const sender = isIA ? 'IA' : isHumanOut ? 'HUMAN' : 'PATIENT';

    if (isIA) hasIA = true;

    if (isIn && !stagesHit.has('CONTATO_INICIAL')) push('CONTATO_INICIAL', 'PATIENT', m);
    if ((isIA || isHumanOut) && greetingRx.test(txt)) push('SAUDACAO', sender, m);
    if ((isIA || isHumanOut) && procedureRx.test(txt)) push('APRESENTACAO_TECNICA', sender, m);
    if (isIn && clinicalRx.test(txt) && txt.length > 3) push('QUEIXA_DECLARADA', 'PATIENT', m);
    if ((isIA || isHumanOut) && prevTreatRx.test(txt)) push('TRIAGEM_CLINICA', sender, m);
    if (priceRx.test(txt)) push('INVESTIMENTO', sender, m);
    if (isIn && objectionRx.test(txt)) push('OBJECAO', 'PATIENT', m);
    if ((isIA || isHumanOut) && contactRx.test(txt)) push('CAPTURA_CONTATO', sender, m);
    if (isHumanOut && hasIA) push('HANDOFF_HUMANO', 'HUMAN', m);
  }

  const outboundIA = messages.filter(m => m.sent_by === 'IA').length;
  const outboundHuman = messages.filter(m => m.direction === 'outbound' && !m.sent_by).length;
  const stages = events.map(e => e.stage);
  const reachedInOrder = stages.filter(s => STAGE_ORDER.includes(s));
  const furthest = reachedInOrder.length > 0
    ? reachedInOrder.reduce((best, s) => STAGE_ORDER.indexOf(s) > STAGE_ORDER.indexOf(best) ? s : best)
    : 'CONTATO_INICIAL';

  return {
    stages,
    furthest,
    track: classifyTrack(outboundIA, outboundHuman),
    outboundIA,
    outboundHuman,
    events
  };
}

function classifyConversations(chatMap, stageConfig) {
  return Object.entries(chatMap).map(([chatId, messages]) => {
    const result = detectStages(messages, stageConfig);
    return {
      chatId,
      msgCount: messages.length,
      inboundCount: messages.filter(m => m.direction === 'inbound').length,
      ...result
    };
  });
}

module.exports = { detectStages, classifyConversations, classifyTrack, STAGE_ORDER };
```

const { STAGE_ORDER } = require('./stage-detector');

function computeMetrics(classified) {
  const total = classified.length;
  const tracks = { pure_ia: { count: 0, with_captura: 0 }, pure_human: { count: 0, with_captura: 0 }, hybrid: { count: 0, with_captura: 0 }, no_outbound: { count: 0, with_captura: 0 } };

  const stageReach = {};
  const stageByIA = {};
  const stageByHuman = {};
  const sequenceCounts = {};
  const anomalies = { price_before_complaint: 0, handoff_before_booking: 0, tech_without_triage: 0 };

  for (const c of classified) {
    const t = c.track || 'no_outbound';
    if (!tracks[t]) tracks[t] = { count: 0, with_captura: 0 };
    tracks[t].count++;
    if (c.stages.includes('CAPTURA_CONTATO')) tracks[t].with_captura++;

    for (const stage of c.stages) {
      stageReach[stage] = (stageReach[stage] || 0) + 1;
    }

    for (const ev of (c.events || [])) {
      if (ev.sender === 'IA') stageByIA[ev.stage] = (stageByIA[ev.stage] || 0) + 1;
      else if (ev.sender === 'HUMAN') stageByHuman[ev.stage] = (stageByHuman[ev.stage] || 0) + 1;
    }

    const seqKey = c.stages.filter(s => STAGE_ORDER.includes(s)).join('>') || 'NONE';
    sequenceCounts[seqKey] = (sequenceCounts[seqKey] || 0) + 1;

    // Anomalies
    const pi = c.stages.indexOf('INVESTIMENTO');
    const pq = c.stages.indexOf('QUEIXA_DECLARADA');
    if (pi !== -1 && (pq === -1 || pi < pq)) anomalies.price_before_complaint++;

    if (c.stages.includes('APRESENTACAO_TECNICA') && !c.stages.includes('TRIAGEM_CLINICA')) anomalies.tech_without_triage++;

    const ph = c.stages.indexOf('HANDOFF_HUMANO');
    const pb = c.stages.indexOf('CAPTURA_CONTATO');
    if (ph !== -1 && (pb === -1 || ph < pb)) anomalies.handoff_before_booking++;
  }

  // Build stages array
  const stages = STAGE_ORDER.map(code => ({
    code,
    reach: stageReach[code] || 0,
    reach_pct: total > 0 ? +((((stageReach[code] || 0) / total) * 100).toFixed(1)) : 0,
    by_ia: stageByIA[code] || 0,
    by_human: stageByHuman[code] || 0
  }));

  // Conversion rates between consecutive STAGE_ORDER stages
  const conversion = [];
  for (let i = 0; i < STAGE_ORDER.length - 1; i++) {
    const from = STAGE_ORDER[i];
    const to = STAGE_ORDER[i + 1];
    const fromCount = stageReach[from] || 0;
    const toCount = stageReach[to] || 0;
    if (fromCount > 0) {
      conversion.push({ from, to, rate_pct: +((toCount / fromCount * 100).toFixed(1)) });
    }
  }

  // Track captura_pct
  for (const [k, v] of Object.entries(tracks)) {
    v.captura_pct = v.count > 0 ? +((v.with_captura / v.count * 100).toFixed(1)) : 0;
  }

  // Top sequences
  const top_sequences = Object.entries(sequenceCounts)
    .sort((a, b) => b[1] - a[1])
    .slice(0, 15)
    .map(([sequence, count]) => ({ sequence, count }));

  return { overview: { total }, tracks, stages, conversion, top_sequences, anomalies };
}

module.exports = { computeMetrics };
```

const BATCH_SIZE = 100;

async function fetchJSON(url, key) {
  const res = await fetch(url, {
    headers: { 'apikey': key, 'Authorization': `Bearer ${key}` }
  });
  if (!res.ok) throw new Error(`Supabase error ${res.status}: ${await res.text()}`);
  return res.json();
}

async function fetchAllChats(accountId) {
  const url = process.env.SUPABASE_URL;
  const key = process.env.SUPABASE_KEY;
  const endpoint = `${url}/rest/v1/wa_chats?account_id=eq.${accountId}&select=id,chat_type,contact_id,created_at,last_message_at&order=last_message_at.desc&limit=2000`;
  return fetchJSON(endpoint, key);
}

async function fetchMessagesForChats(chatIds) {
  const url = process.env.SUPABASE_URL;
  const key = process.env.SUPABASE_KEY;
  const allMessages = [];

  for (let i = 0; i < chatIds.length; i += BATCH_SIZE) {
    const batch = chatIds.slice(i, i + BATCH_SIZE);
    const filter = batch.join(',');
    const endpoint = `${url}/rest/v1/wa_messages?chat_id=in.(${filter})&select=id,chat_id,direction,sent_by,content_text,created_at,message_type&order=created_at.asc&limit=10000`;
    const msgs = await fetchJSON(endpoint, key);
    allMessages.push(...msgs);
  }

  return allMessages;
}

function groupMessagesByChat(messages) {
  const map = {};
  for (const m of messages) {
    if (!map[m.chat_id]) map[m.chat_id] = [];
    map[m.chat_id].push(m);
  }
  return map;
}

async function fetchConversations(accountId) {
  const chats = await fetchAllChats(accountId);
  const chatIds = chats.map(c => c.id);
  const messages = await fetchMessagesForChats(chatIds);
  const messagesByChat = groupMessagesByChat(messages);
  return { chats, messagesByChat, totalMessages: messages.length };
}

module.exports = { fetchConversations };
```

const OpenAI = require('openai');
const cache = require('./cache');

const SYSTEM_PROMPT = `You are a funnel stage configuration extractor.
Read the AI assistant prompt below and return a JSON object with these exact fields:
{
  "clinical_terms": string[],        // body parts / conditions patients might mention
  "procedure_terms": string[],       // procedure names the assistant discusses
  "greeting_patterns": string[],     // opening words the assistant uses to greet
  "previous_treatment_terms": string[], // treatments the assistant asks patients about
  "price_pattern": string,           // regex pattern for prices (e.g. "R\\\\$\\\\s*[\\\\d.,]+\\\\s*(mil|k)?")
  "objection_terms": string[],       // words patients use when objecting (distance, fear, cost)
  "contact_capture_terms": string[], // phrases used when capturing patient contact info
  "forbidden_words": string[],       // words explicitly prohibited in the prompt
  "handoff_triggers": string[]       // phrases signaling patient wants to book/schedule
}
Output ONLY valid JSON. No prose, no markdown fences.`;

async function parsePrompt(promptText) {
  const hash = cache.hashPrompt(promptText);
  const cached = cache.get(hash);
  if (cached) return { stageConfig: cached, cacheHit: true, hash };

  const client = new OpenAI({ apiKey: process.env.OPENAI_API_KEY });
  const response = await client.chat.completions.create({
    model: 'gpt-4o-mini',
    messages: [
      { role: 'system', content: SYSTEM_PROMPT },
      { role: 'user', content: promptText }
    ],
    response_format: { type: 'json_object' },
    temperature: 0
  });

  const stageConfig = JSON.parse(response.choices[0].message.content);
  cache.set(hash, stageConfig);
  return { stageConfig, cacheHit: false, hash };
}

module.exports = { parsePrompt };
```

  });

require('dotenv').config();
const express = require('express');
const { fetchConversations } = require('./supabase');
const { isProfessional } = require('./filter');
const { classifyConversations } = require('./stage-detector');
const { computeMetrics } = require('./metrics');
const { parsePrompt } = require('./prompt-parser');
const { generateReport } = require('./report-writer');
const cache = require('./cache');

const app = express();
app.use(express.json({ limit: '2mb' }));

const UUID_RE = /^[0-9a-f]{8}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{12}$/i;

app.get('/health', (_req, res) => res.json({ ok: true }));

app.get('/cache', (_req, res) => res.json(cache.list()));

app.delete('/cache/:hash', (req, res) => {
  cache.del(req.params.hash);
  res.json({ deleted: req.params.hash });
});

app.post('/analyze', async (req, res) => {
  const { account_id, prompt } = req.body || {};

  if (!account_id || !UUID_RE.test(account_id))
    return res.status(400).json({ error: 'account_id must be a valid UUID' });
  if (!prompt || typeof prompt !== 'string' || prompt.trim().length === 0)
    return res.status(400).json({ error: 'prompt must be a non-empty string' });

  try {
    // Step 1: parse prompt + fetch data in parallel
    const [{ stageConfig, cacheHit, hash }, { chats, messagesByChat, totalMessages }] =
      await Promise.all([parsePrompt(prompt), fetchConversations(account_id)]);

    // Step 2: filter professional conversations
    const professionalChatMap = {};
    let personalCount = 0;
    for (const [chatId, msgs] of Object.entries(messagesByChat)) {
      if (isProfessional(msgs, stageConfig)) professionalChatMap[chatId] = msgs;
      else personalCount++;
    }

    // Step 3: classify stages
    const classified = classifyConversations(professionalChatMap, stageConfig);

    // Step 4: compute metrics
    const metrics = computeMetrics(classified);

    // Step 5: generate report
    const report_md = await generateReport(metrics, account_id);

    res.json({
      meta: {
        account_id,
        analyzed_at: new Date().toISOString(),
        total_chats: chats.length,
        professional: classified.length,
        personal: personalCount,
        total_messages: totalMessages,
        prompt_cache_hit: cacheHit,
        prompt_hash: hash
      },
      funnel: metrics,
      stage_config: stageConfig,
      report_md
    });
  } catch (err) {
    console.error(err);
    res.status(500).json({ error: err.message });
  }
});

const PORT = process.env.PORT || 3000;
if (require.main === module) {
  app.listen(PORT, () => console.log(`funnel-analyzer listening on http://localhost:${PORT}`));
}

module.exports = app;
```

## How It Works

1. Prompt text is parsed by `gpt-4o-mini` once → result cached by MD5 hash (no repeat LLM cost)
2. All conversations fetched from Supabase in batches of 100
3. Each conversation classified as professional or personal (3-tier rule engine)
4. Professional conversations tagged with funnel stage events (pure regex)
5. Metrics computed (reach %, conversion rates, track split, anomalies)
6. `gpt-4o-mini` generates the markdown narrative from the computed metrics
